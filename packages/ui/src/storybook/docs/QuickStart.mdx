# Quick Start Guide

Get started quickly with common patterns and examples.

## Installation

```bash
# Install the UI package
pnpm install @design-studio/ui

# Or with npm
npm install @design-studio/ui
```

## Basic Setup

### 1. Import Styles

```tsx
import "@design-studio/ui/styles";
```

Or in your CSS:

```css
@import "@design-studio/ui/styles";
```

### 2. Import Components

```tsx
import { Button, ChatHeader, ChatInput } from "@design-studio/ui";
```

## Common Patterns

### Pattern 1: Simple Button

```tsx
import { Button } from "@design-studio/ui";

export function Example() {
  return (
    <Button variant="primary" onClick={() => alert("Clicked!")}>
      Click Me
    </Button>
  );
}
```

### Pattern 2: Chat Interface

```tsx
import { ChatHeader, ChatSidebar, ChatMessages, ChatInput } from "@design-studio/ui";

export function ChatApp() {
  const [isSidebarOpen, setIsSidebarOpen] = useState(true);
  const [messages, setMessages] = useState<ChatMessage[]>([]);
  const [selectedModel, setSelectedModel] = useState<ModelConfig>({
    name: "ChatGPT 5.2 Pro",
    shortName: "5.2 Pro",
    description: "Our most capable model",
  });

  return (
    <>
      <ChatSidebar isOpen={isSidebarOpen} onToggle={() => setIsSidebarOpen(!isSidebarOpen)} />
      <ChatHeader
        isSidebarOpen={isSidebarOpen}
        onSidebarToggle={() => setIsSidebarOpen(!isSidebarOpen)}
        selectedModel={selectedModel}
        onModelChange={setSelectedModel}
      />
      <ChatMessages messages={messages} />
      <ChatInput
        selectedModel={selectedModel}
        onSend={(content) => {
          setMessages([
            ...messages,
            {
              id: Date.now().toString(),
              role: "user",
              content,
              timestamp: new Date(),
            },
          ]);
        }}
      />
    </>
  );
}
```

### Pattern 3: Settings Panel

```tsx
import { SettingRow, SettingToggle } from "@design-studio/ui";

export function SettingsPanel() {
  const [darkMode, setDarkMode] = useState(false);
  const [notifications, setNotifications] = useState(true);

    <div className="space-y-4">
      <SettingRow label="Dark Mode" description="Use dark theme across the app">

      <SettingRow label="Notifications" description="Enable push notifications">
        <SettingToggle checked={notifications} onCheckedChange={setNotifications} />
      </SettingRow>
    </div>
  );
}
```

### Pattern 4: Form with Validation

```tsx
import { Input, Button } from "@design-studio/ui";

export function ContactForm() {
  const [email, setEmail] = useState("");
  const [isValid, setIsValid] = useState(true);

  const validateEmail = (value: string) => {
    return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value);
  };

  const handleSubmit = () => {
    if (validateEmail(email)) {
      // Submit form
      alert("Submitted: " + email);
    } else {
      setIsValid(false);
    }
  };

  return (
    <form onSubmit={(e) => e.preventDefault()}>
      <Input
        type="email"
        placeholder="your@email.com"
        value={email}
        onChange={(e) => {
          setEmail(e.target.value);
          setIsValid(true);
        }}
        className={!isValid ? "border-red-500" : ""}
      />
      <Button onClick={handleSubmit}>Submit</Button>
    </form>
  );
}
```

### Pattern 5: Modal Dialog

```tsx
import { Dialog, DialogContent, DialogHeader, DialogTitle } from "@design-studio/ui";

export function Example() {
  const [isOpen, setIsOpen] = useState(false);

  return (
    <>
      <Button onClick={() => setIsOpen(true)}>Open Dialog</Button>

      <Dialog open={isOpen} onOpenChange={setIsOpen}>
        <DialogContent>
          <DialogHeader>
            <DialogTitle>Confirm Action</DialogTitle>
          </DialogHeader>
          <p>Are you sure you want to proceed?</p>
          <div className="flex gap-2 justify-end">
            <Button variant="ghost" onClick={() => setIsOpen(false)}>
              Cancel
            </Button>
            <Button variant="destructive" onClick={() => setIsOpen(false)}>
              Confirm
            </Button>
          </div>
        </DialogContent>
      </Dialog>
    </>
  );
}
```

### Pattern 6: Tabbed Interface

```tsx
import { Tabs, TabsContent, TabsList, TabsTrigger } from "@design-studio/ui";

export function TabbedExample() {
  return (
    <Tabs defaultValue="tab1">
      <TabsList>
        <TabsTrigger value="tab1">Tab 1</TabsTrigger>
        <TabsTrigger value="tab2">Tab 2</TabsTrigger>
        <TabsTrigger value="tab3">Tab 3</TabsTrigger>
      </TabsList>
      <TabsContent value="tab1">Content for tab 1</TabsContent>
      <TabsContent value="tab2">Content for tab 2</TabsContent>
      <TabsContent value="tab3">Content for tab 3</TabsContent>
    </Tabs>
  );
}
```

### Pattern 7: Toast Notification

```tsx
import { Toast } from "@design-studio/ui";

export function ToastExample() {
  const [showToast, setShowToast] = useState(false);

  return (
    <>
      <Button onClick={() => setShowToast(true)}>Show Toast</Button>

      {showToast && (
        <Toast
          variant="success"
          title="Success!"
          duration={3000}
          onClose={() => setShowToast(false)}
        >
          Your changes have been saved.
        </Toast>
      )}
    </>
  );
}
```

### Pattern 8: Full Template Layout

```tsx
import { DashboardTemplate } from "@design-studio/ui";
import { ChatHeader } from "@design-studio/ui/chat";
import { ChatSidebar } from "@design-studio/ui/chat";
import { ChatMessages } from "@design-studio/ui/chat";
import { ChatInput } from "@design-studio/ui/chat";

export function App() {
  const [isSidebarOpen, setIsSidebarOpen] = useState(true);

  return (
    <DashboardTemplate
      header={
        <ChatHeader
          isSidebarOpen={isSidebarOpen}
          onSidebarToggle={() => setIsSidebarOpen(!isSidebarOpen)}
        />
      }
      sidebar={
        isSidebarOpen ? (
          <ChatSidebar isOpen={isSidebarOpen} onToggle={() => setIsSidebarOpen(!isSidebarOpen)} />
        ) : null
      }
      body={<ChatMessages />}
      footer={<ChatInput />}
    />
  );
}
```

## Component Categories

### By Task

| Task             | Use This Component                |
| ---------------- | --------------------------------- |
| **Display data** | Card, Table, Chart                |
| **Get input**    | Input, Textarea, Select, Combobox |
| **Navigate**     | Tabs, Breadcrumb, Sidebar         |
| **Show status**  | Badge, Alert, Toast, Progress     |
| **Actions**      | Button, Toggle, Checkbox, Radio   |
| **Layouts**      | Separator, ScrollArea, Resizable  |
| **Overlays**     | Dialog, Modal, Tooltip, Popover   |

### By Complexity

| Level         | Components                      |
| ------------- | ------------------------------- |
| **Basic**     | Button, Input, Badge, Text      |
| **Composite** | Form, SettingRow, ChatInput     |
| **Layout**    | DashboardTemplate, ChatTemplate |
| **Feature**   | ChatShell, SettingsModal        |

## Styling

### Using Design Tokens

```tsx
import { FColor, FSpacing, FType } from "@design-studio/ui/foundation";

// Tokens are available as CSS variables
const styles = {
  color: "var(--foundation-text-primary)",
  padding: "var(--spacing-s16)",
  fontSize: "var(--type-body-medium)",
};
```

### Custom Styling

```tsx
<Button className="custom-button">
  Click Me
</Button>

<style>
.custom-button {
  background: var(--foundation-accent-blue);
  color: var(--foundation-text-primary);
}
</style>
```
